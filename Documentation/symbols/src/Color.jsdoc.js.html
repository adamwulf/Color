 42 	var hue2rgb = function(a, b, c) {  // http://www.w3.org/TR/css3-color/#hsl-color
 43 		if(c < 0) c += 1;
 44 		if(c > 1) c -= 1;
 45 		if(c < 1/6) return a + (b - a) * 6 * c;
 46 		if(c < 1/2) return b;
 47 		if(c < 2/3) return a + (b - a) * (2/3 - c) * 6;
 48 		return a;
 49 	};

 67 	/**
 68 	* Color instance - get, update and output a Color between structures.
 69 	* @constructor
 70 	* @param {mixed} value Accepts any valid CSS color value e.g., #FF9900, rgb(255, 153, 0), rgba(100%, 40%, 0%, 0.8);
 71 	* a hash with properties mapped to the Color instance e.g., red, green, saturation, brightness;
 72 	* another Color instance; a numeric color value; a named CSS color
 73 	* @class Instances of the Color class serve as abstract reprentations of the color itself, and don't need to be 
 74 	* transformed from one format to another.  A single Color instance can have any component (red, green, blue, hue, saturation, lightness, brightness,
 75 	* alpha) updated regardless of the source.  Further, all other components will be normalized automatically.  If a Color is instanced using a hex value,
 76 	* it can have it's lightness component updated directly despite lightness being a HSL component.  Further, the same color instance can output it's
 77 	* component values in any format without any extra conversions.  Conversion methods (getRGB, getHex) are provided just as helpers, and don't perform any
 78 	* actual transformations.  They are not required for use or translation.  The standard component parts are available as instance methods - passing a value
 79 	* argument to set, and each return the value as well (with or without setter arguments).  These components perform transformations and dispatch events, and
 80 	* can be used without any sugar to manage the Color instance.
 81 	* <br />
 82 	* Component methods include:
 83 	* <ul>
 84 	* <li>.red()</li>
 85 	* <li>.green()</li>
 86 	* <li>.blue()</li>
 87 	* <li>.hue()</li>
 88 	* <li>.saturation()</li>
 89 	* <li>.lightness()</li>
 90 	* <li>.brightness()</li>
 91 	* <li>.hex()</li>
 92 	* <li>.value()</li>
 93 	* </ul>
 94 	* @example
 95 	// instancing...
 96 	* new Color();
 97 	* new Color('#FF9900');
 98 	* new Color(element.style.color);
 99 	* new Color('pink');
100 	* new Color(123456);
101 	* new Color({ red : 255, green : 100, blue : 0 });
102 	* new Color(colorInstance);
103 	* // usage...
104 	* var color = new Color('#FF9900');
105 	* color.brightness(20);
106 	* element.style.backgroundColor = color;
107 	* console.log(color.getRGB());
108 	* console.log(color.saturation());
109 	*/

135 	/**
136 	* Convert mixed variable to Color component properties, and adopt those properties.
137 	* @function
138 	* @param {mixed} value Accepts any valid CSS color value e.g., #FF9900, rgb(255, 153, 0), rgba(100%, 40%, 0%, 0.8);
139 	* a hash with properties mapped to the Color instance e.g., red, green, saturation, brightness;
140 	* another Color instance; a numeric color value; a named CSS color
141 	* @returns this
142 	* @example
143 	* var color = new Color();
144 	* color.parse();
145 	* color.parse('#FF9900');
146 	* color.parse(element.style.color);
147 	* color.parse('pink');
148 	* color.parse(123456);
149 	* color.parse({ red : 255, green : 100, blue : 0 });
150 	* color.parse(colorInstance);
151 	*/
152 	Color.prototype.parse = function(value){
153 		if(typeof value == 'undefined'){
154 			return this;
155 		};
156 		switch(true){
157 			case isFinite(value) :
158 				return this.value(value);
159 			case (value instanceof Color) :
160 				return this.copy(value);
161 			case isHex.test(value) :
162 				var stripped = value.replace(leadHex, '');
163 				if(stripped.length == 3) {
164 					stripped = stripped.replace(hexBit, '$1$1');
165 				};
166 				return this.value(parseInt(stripped, 16));
167 			case isRGB.test(value) :
168 				var parts = value.match(matchRGB);
169 				this.red(perc2value(parts[1]));
170 				this.green(perc2value(parts[2]));
171 				this.blue(perc2value(parts[3]));
172 				this.alpha(parseInt(parts[4]) || 1);
173 				return this.value();
174 			case isHSL.test(value) :  // http://www.w3.org/TR/css3-color/#hsl-color
175 				var parts = value.match(matchHSL);
176 				this.hue(parseInt(parts[1]));
177 				this.saturation(parseInt(parts[2]));
178 				this.lightness(parseInt(parts[3]));
179 				this.alpha(parseInt(parts[4]) || 1);
180 				return this.value();
181 			case (namedColors.hasOwnProperty(value)) :
182 				value = namedColors[value];
183 				var stripped = value.replace(leadHex, '');
184 				return this.value(parseInt(stripped, 16));
185 			case (typeof value == 'object') :
186 				return this.set(value);
187 		};
188 		return this;
189 	};

249 	* Create a new Color instance with values between the invoking Color instance and the destination Color instance
250 	* @function
251 	* @param {Color} destination Color instance to serve as the termination of the interpolation
252 	* @param {number} factor 0-1, where 0 is the origin Color and 1 is the destination Color, and 0.5 is halfway between.  This method will "blend" the colors.
253 	* @returns this
254 	* @example
255 	* var color = new Color();
256 	* color.set('lightness', 100);
257 	* color.set({ red : 255, green : 100 });
258 	* color.set(123456);

260 	Color.prototype.interpolate = function(destination, factor){
261 		if(!(destination instanceof Color)){
262 			destination = new Color(destination);
263 		};
264 		this._red = this._red + (destination._red - this._red) * factor;
265 		this._green = this._green + (destination._green - this._green) * factor;
266 		this._blue = this._blue + (destination._blue - this._blue) * factor;
267 		this.broadcast(Event.RGB_UPDATED);
268 		return this;
269 	};

271 	Color.prototype._RGB2HSL = function(){
272 
273 		var r = this._red / 255;
274 		var g = this._green / 255;
275 		var b = this._blue / 255;
276 
277 		var max = Math.max(r, g, b);
278 		var min = Math.min(r, g, b);
279 		var l = (max + min) / 2;
280 		var v = max;
281 
282 		if(max == min) {
283 			this._hue = 0;
284 			this._saturation = 0;
285 			this._lightness = absround(l * 100);
286 			this._brightness = absround(v * 100);
287 			return;
288 		};
289 
290 		var d = max - min;
291 		var s = d / ( ( l <= 0.5) ? (max + min) : (2 - max - min) );
292 		var h = ((max == r)
293 			? (g - b) / d + (g < b ? 6 : 0)
294 			: (max == g)
295 			 ? ((b - r) / d + 2)
296 			 : ((r - g) / d + 4)) / 6;
297 
298 		this._hue = absround(h * 360);
299 		this._saturation = absround(s * 100);
300 		this._lightness = absround(l * 100);
301 		this._brightness = absround(v * 100);
302 	};
303 	Color.prototype._HSL2RGB = function(){
304 		var h = this._hue / 360;
305 		var s = this._saturation / 100;
306 		var l = this._lightness / 100;
307 		var q = l < 0.5	? l * (1 + s) : (l + s - l * s);
308 		var p = 2 * l - q;
309 		this._red = absround(hue2rgb(p, q, h + 1/3) * 255);
310 		this._green = absround(hue2rgb(p, q, h) * 255);
311 		this._blue = absround(hue2rgb(p, q, h - 1/3) * 255);
312 	};
313 	Color.prototype._HSV2RGB = function(){  // http://mjijackson.com/2008/02/rgb-to-hsl-and-rgb-to-hsv-color-model-conversion-algorithms-in-javascript
314 		var h = this._hue / 360;
315 		var s = this._saturation / 100;
316 		var v = this._brightness / 100;
317 		var r = 0;
318 		var g = 0;
319 		var b = 0;
320 		var i = absround(h * 6);
321 		var f = h * 6 - i;
322 		var p = v * (1 - s);
323 		var q = v * (1 - f * s);
324 		var t = v * (1 - (1 - f) * s);
325 		switch(i % 6){
326 			case 0 :
327 				r = v, g = t, b = p;
328 				break;
329 			case 1 :
330 				r = q, g = v, b = p;
331 				break;
332 			case 2 :
333 				r = p, g = v, b = t;
334 				break;
335 			case 3 :
336 				r = p, g = q, b = v;
337 				break;
338 			case 4 :
339 				r = t, g = p, b = v
340 				break;
341 			case 5 :
342 				r = v, g = p, b = q;
343 				break;
344 		}
345 		this._red = absround(r * 255);
346 		this._green = absround(g * 255);
347 		this._blue = absround(b * 255);
348 	};

624 	* Returns a tokenized string from the Color's component values
625 	* @function
626 	* @param {string} string The string to return, with tokens expressed as %token% that are replaced with component values.  Tokens are as follows:
627 	* <ul>
628 	* <li>r : red</li>
629 	* <li>g : green</li>
630 	* <li>b : blue</li>
631 	* <li>h : hue</li>
632 	* <li>s : saturation</li>
633 	* <li>l : lightness</li>
634 	* <li>v : brightness</li>
635 	* <li>a : alpha</li>
636 	* <li>x : hex</li>
637 	* <li>i : value</li>
638 	* </ul>
639 	* @returns String
640 	* @example
641 	* var color = new Color('#FF9900');
642 	* console.log(color.format('red=%r%, green=%g%, blue=%b%));
643 	*/
644 	Color.prototype.format = function(string){
645 		var tokens = {
646 			r : this._red,
647 			g : this._green,
648 			b : this._blue,
649 			h : this._hue,
650 			s : this._saturation,
651 			l : this._lightness,
652 			v : this._brightness,
653 			a : this._alpha,
654 			x : this._hex,
655 			i : this._value
656 		};
657 		for(var token in tokens){
658 			string = string.split('%' + token + '%').join(tokens[token]);
659 		};
660 		return string;
661 	};

709 	// Event Management
710 	Color.prototype._listeners = null;
711 	Color.prototype._isSubscribed = function(type){
712 		return this._listeners[type] != null;
713 	};
714 	/**
715 	* @function
716 	* @param {string} type Event type to listen for
717 	* @param {function} callback listener to register to the event
718 	* @example 
719 	* var color = new Color();
720 	* color.subscribe(Color.Event.UPDATED, function(){
721 	*   alert('this color has been updated');
722 	* });
723 	* color.red(255);
724 	*/
725 	Color.prototype.subscribe = function(type, callback){
726 		if(!this._isSubscribed(type)) {
727 			this._listeners[type] = [];
728 		};
729 		this._listeners[type].push(callback);
730 	};
731 	/**
732 	* @function
733 	* @param {string} type Event type to remove the listener from
734 	* @param {function} callback listener to unregister from the event
735 	* @example 
736 	* var color = new Color();
737 	* var handler = function(){
738 	*   console.log('this color has been updated');
739 	* });
740 	* color.subscribe(Color.Event.UPDATED, handler);
741 	* color.red(255);
742 	* color.unsubscribe(Color.Event.UPDATED, handler);
743 	* color.red(0);
744 	*/
745 	Color.prototype.unsubscribe = function(type, callback){
746 		if(!this._isSubscribed(type)) {
747 			return;
748 		};
749 		var stack = this._listeners[type];
750 		var len = stack.length;
751 		for(var i = 0; i < l; i++){
752 			if(stack[i] === callback){
753 				stack.splice(i, 1);
754 				return this.unsubscribe(type, callback);
755 			};
756 		};
757 	};
758 	/**
759 	* @function
760 	* @param {string} type Event type to dispatch
761 	* @param {array} params Array of arguments to pass to listener
762 	* @example 
763 	* var color = new Color();
764 	* var handler = function(a, b){
765 	*   console.log('a=' + a, 'b=' + b);
766 	* });
767 	* color.subscribe('arbitraryEvent', handler);
768 	* color.broadcast('arbitraryEvent', ['A', 'B']);
769 	*/
770 	Color.prototype.broadcast = function(type, params){
771 		if(!this._isSubscribed(type)) {
772 			return;
773 		}
774 		var stack = this._listeners[type];
775 		var l = stack.length;
776 		for(var i = 0; i < l; i++) {
777 			stack[i].apply(this, params);
778 		}
779 	};
