 41 	var hue2rgb = function(a, b, c) {  // http://www.w3.org/TR/css3-color/#hsl-color
 42 		if(c < 0) c += 1;
 43 		if(c > 1) c -= 1;
 44 		if(c < 1/6) return a + (b - a) * 6 * c;
 45 		if(c < 1/2) return b;
 46 		if(c < 2/3) return a + (b - a) * (2/3 - c) * 6;
 47 		return a;
 48 	};

 95 	/**
 96 	* Convert mixed variable to Color component properties, and adopt those properties.
 97 	* @function
 98 	* @param {mixed} value Accepts any valid CSS color value e.g., #FF9900, rgb(255, 153, 0), rgba(100%, 40%, 0%, 0.8);
 99 	* a hash with properties mapped to the Color instance e.g., red, green, saturation, brightness;
100 	* another Color instance; a numeric color value; a named CSS color
101 	* @returns this
102 	* @example
103 	* var color = new Color();
104 	* color.parse();
105 	* color.parse('#FF9900');
106 	* color.parse(element.style.color);
107 	* color.parse('pink');
108 	* color.parse(123456);
109 	* color.parse({ red : 255, green : 100, blue : 0 });
110 	* color.parse(colorInstance);
111 	*/
112 	Color.prototype.parse = function(value){
113 		if(typeof value == 'undefined'){
114 			return this;
115 		};
116 		switch(true){
117 			case isFinite(value) :
118 				return this.value(value);
119 			case (value instanceof Color) :
120 				return this.copy(value);
121 			case isHex.test(value) :
122 				var stripped = value.replace(leadHex, '');
123 				if(stripped.length == 3) {
124 					stripped = stripped.replace(hexBit, '$1$1');
125 				};
126 				return this.value(parseInt(stripped, 16));
127 			case isRGB.test(value) :
128 				var parts = value.match(matchRGB);
129 				this.red(perc2value(parts[1]));
130 				this.green(perc2value(parts[2]));
131 				this.blue(perc2value(parts[3]));
132 				this.alpha(parseInt(parts[4]) || 1);
133 				return this.value();
134 			case isHSL.test(value) :  // http://www.w3.org/TR/css3-color/#hsl-color
135 				var parts = value.match(matchHSL);
136 				this.hue(parseInt(parts[1]));
137 				this.saturation(parseInt(parts[2]));
138 				this.lightness(parseInt(parts[3]));
139 				this.alpha(parseInt(parts[4]) || 1);
140 				return this.value();
141 			case (namedColors.hasOwnProperty(value)) :
142 				value = namedColors[value];
143 				var stripped = value.replace(leadHex, '');
144 				return this.value(parseInt(stripped, 16));
145 			case (typeof value == 'object') :
146 				return this.set(value);
147 		};
148 		return this;
149 	};

209 	* Create a new Color instance with values between the invoking Color instance and the destination Color instance
210 	* @function
211 	* @param {string|object|number} destination Color instance to serve as the termination of the interpolation
212 	* @param {string|number} factor 0-1, where 0 is the origin Color and 1 is the destination Color, and 0.5 is halfway between.  This method will "blend" the colors.
213 	* @returns this
214 	* @example
215 	* var color = new Color();
216 	* color.set('lightness', 100);
217 	* color.set({ red : 255, green : 100 });
218 	* color.set(123456);
219 	*/
220 	Color.prototype.interpolate = function(destination, factor){
221 		if(!(destination instanceof Color)){
222 			destination = new Color(destination);
223 		};
224 		this._red = this._red + (destination._red - this._red) * factor;
225 		this._green = this._green + (destination._green - this._green) * factor;
226 		this._blue = this._blue + (destination._blue - this._blue) * factor;
227 		this.broadcast(Event.RGB_UPDATED);
228 		return this;
229 	};

231 	Color.prototype.RGB2HSL = function(){
232 
233 		var r = this._red / 255;
234 		var g = this._green / 255;
235 		var b = this._blue / 255;
236 
237 		var max = Math.max(r, g, b);
238 		var min = Math.min(r, g, b);
239 		var l = (max + min) / 2;
240 		var v = max;
241 
242 		if(max == min) {
243 			this._hue = 0;
244 			this._saturation = 0;
245 			this._lightness = absround(l * 100);
246 			this._brightness = absround(v * 100);
247 			return;
248 		};
249 
250 		var d = max - min;
251 		var s = d / ( ( l <= 0.5) ? (max + min) : (2 - max - min) );
252 		var h = ((max == r)
253 			? (g - b) / d + (g < b ? 6 : 0)
254 			: (max == g)
255 			 ? ((b - r) / d + 2)
256 			 : ((r - g) / d + 4)) / 6;
257 
258 		this._hue = absround(h * 360);
259 		this._saturation = absround(s * 100);
260 		this._lightness = absround(l * 100);
261 		this._brightness = absround(v * 100);
262 	};
263 	Color.prototype.HSL2RGB = function(){
264 		var h = this._hue / 360;
265 		var s = this._saturation / 100;
266 		var l = this._lightness / 100;
267 		var q = l < 0.5	? l * (1 + s) : (l + s - l * s);
268 		var p = 2 * l - q;
269 		this._red = absround(hue2rgb(p, q, h + 1/3) * 255);
270 		this._green = absround(hue2rgb(p, q, h) * 255);
271 		this._blue = absround(hue2rgb(p, q, h - 1/3) * 255);
272 	};
273 	Color.prototype.HSV2RGB = function(){  // http://mjijackson.com/2008/02/rgb-to-hsl-and-rgb-to-hsv-color-model-conversion-algorithms-in-javascript
274 		var h = this._hue / 360;
275 		var s = this._saturation / 100;
276 		var v = this._brightness / 100;
277 		var r = 0;
278 		var g = 0;
279 		var b = 0;
280 		var i = absround(h * 6);
281 		var f = h * 6 - i;
282 		var p = v * (1 - s);
283 		var q = v * (1 - f * s);
284 		var t = v * (1 - (1 - f) * s);
285 		switch(i % 6){
286 			case 0 :
287 				r = v, g = t, b = p;
288 				break;
289 			case 1 :
290 				r = q, g = v, b = p;
291 				break;
292 			case 2 :
293 				r = p, g = v, b = t;
294 				break;
295 			case 3 :
296 				r = p, g = q, b = v;
297 				break;
298 			case 4 :
299 				r = t, g = p, b = v
300 				break;
301 			case 5 :
302 				r = v, g = p, b = q;
303 				break;
304 		}
305 		this._red = absround(r * 255);
306 		this._green = absround(g * 255);
307 		this._blue = absround(b * 255);
308 	};

642 		return this.listeners[type] != null;
643 	};
644 	Color.prototype.subscribe = function(type, callback){
645 		if(!this.isSubscribed(type)) {
646 			this.listeners[type] = [];
647 		};
648 		this.listeners[type].push(callback);
649 	};
650 	Color.prototype.unsubscribe = function(type, callback){
651 		if(!this.isSubscribed(type)) {
652 			return;
653 		};
654 		var stack = this.listeners[type];
655 		var len = stack.length;
656 		for(var i = 0; i < l; i++){
657 			if(stack[i] === callback){
658 				stack.splice(i, 1);
659 				return this.unsubscribe(type, callback);
660 			};
661 		};
662 	};
663 	Color.prototype.broadcast = function(type, params){
664 		if(!this.isSubscribed(type)) {
665 			return;
666 		}
667 		var stack = this.listeners[type];
668 		var l = stack.length;
669 		for(var i = 0; i < l; i++) {
670 			stack[i].apply(this, params);
671 		}
672 	};
